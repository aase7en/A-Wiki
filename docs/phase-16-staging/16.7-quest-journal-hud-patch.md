## Ticket Goal

Surface the quest journal (16.6) as a "สมุดเป้าหมาย" compact parchment panel in the HUD.
This panel doubles as:

- **First-run onboarding**: new players see a clear "what do I do today" guide on day 1
- **Progression spine**: daily quests and milestone badges give structured feedback loops
- **Economy feedback**: milestones anchor to real `netWorth`, `botPnl`, and `liabilities` from 16.1–16.5

Layout is mobile-safe (min 390px), uses Thai text throughout (≥14px, line-height 1.65), and
follows the existing parchment visual language of Debt Dungeon and other panels.

## 2. New Component — `src/components/QuestJournalPanel.tsx`

```tsx
// src/components/QuestJournalPanel.tsx
// Ticket 16.7 — "สมุดเป้าหมาย" HUD panel
// Depends on: Ticket 16.6 (questJournal.ts), Ticket 16.3 (netWorth in store)

import React from 'react';
import { usePwqStore } from '../state/store';
import {
  buildDailyJournal,
  type JournalInputs,
  type Quest,
  type Milestone,
} from '../logic/questJournal';

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

function QuestRow({ quest }: { quest: Quest }) {
  const pct = quest.goal > 0
    ? Math.round((quest.progress / quest.goal) * 100)
    : 0;

  return (
    <div className="quest-row pwq-finance-readable" aria-label={quest.nameTh}>
      <div className="quest-row-header">
        <span className="quest-name">{quest.nameTh}</span>
        {quest.completed && (
          <span className="quest-checkmark" aria-label="เสร็จแล้ว">✅</span>
        )}
      </div>
      <div className="quest-desc">{quest.descTh}</div>
      <div
        className="quest-progress-bar"
        role="progressbar"
        aria-valuenow={quest.progress}
        aria-valuemin={0}
        aria-valuemax={quest.goal}
        aria-label={`${quest.nameTh}: ${quest.progress} / ${quest.goal}`}
      >
        <div
          className="quest-progress-fill"
          style={{ width: `${pct}%` }}
        />
      </div>
      <div className="quest-progress-label">
        {quest.progress} / {quest.goal}
      </div>
      {!quest.completed && (
        <div className="quest-hint">{quest.nextActionHint}</div>
      )}
    </div>
  );
}

function MilestoneBadge({ milestone }: { milestone: Milestone }) {
  return (
    <div
      className={`milestone-badge${milestone.unlocked ? ' unlocked' : ''}`}
      aria-label={`${milestone.nameTh}: ${milestone.unlocked ? 'ปลดล็อกแล้ว' : 'ยังไม่ถึง'}`}
    >
      <span className="milestone-icon" aria-hidden="true">
        {milestone.unlocked ? '🏅' : '🔒'}
      </span>
      <span className="milestone-name">{milestone.nameTh}</span>
    </div>
  );
}

// ---------------------------------------------------------------------------
// Main component
// ---------------------------------------------------------------------------

export function QuestJournalPanel() {
  const openModal      = usePwqStore((s) => s.openModal);
  const setOpenModal   = usePwqStore((s) => s.setOpenModal);
  const currentDay     = usePwqStore((s) => s.currentDay);

  // Economy fields (16.1–16.3)
  const coins          = usePwqStore((s) => s.coins);
  const netWorth       = usePwqStore((s) => s.netWorth);
  const netWorthHistory = usePwqStore((s) => s.netWorthHistory);
  const botStakes      = usePwqStore((s) => s.botStakes);
  const liabilities    = usePwqStore((s) => s.liabilities);
  const lastBotPnl     = usePwqStore((s) => s.lastBotPnl);

  // Daily activity counters
  const plotsHarvested    = usePwqStore((s) => s.plotsHarvested);
  const plotsWatered      = usePwqStore((s) => s.plotsWatered);
  const plotsSeeded       = usePwqStore((s) => s.plotsSeeded);
  const coinsShipped      = usePwqStore((s) => s.coinsShipped);
  const animalsFed        = usePwqStore((s) => s.animalsFed);
  const npcTalkedToday    = usePwqStore((s) => s.npcTalkedToday);
  const analystDeskOpened = usePwqStore((s) => s.analystDeskOpened);
  const isFestivalDay     = usePwqStore((s) => s.isFestivalDay);
  const milestonesUnlocked = usePwqStore((s) => s.milestonesUnlocked);

  if (openModal !== 'quest-journal') return null;

  // Build journal data — pure call, no side effects
  const journalInputs: JournalInputs = {
    coins,
    netWorth,
    netWorthHistory,
    botStakes,
    liabilities,
    lastBotPnl,
    plotsHarvested,
    plotsWatered,
    plotsSeeded,
    coinsShipped,
    animalsFed,
    npcTalkedToday,
    analystDeskOpened,
    isFestivalDay,
    milestonesUnlocked,
  };
  const journal = buildDailyJournal(journalInputs, currentDay);

  const anyCompleted = journal.quests.some((q) => q.completed);

  // All 4 milestones (include already-unlocked ones for badge display).
  // checkMilestones returns only *newly* unlocked; for display we want all.
  // We reconstruct from the union of already-unlocked + newly unlocked.
  const ALL_MILESTONE_DEFS: Array<{ id: string; nameTh: string }> = [
    { id: 'milestone-10k-networth',   nameTh: 'มูลค่าสุทธิ ฿10,000 แรก' },
    { id: 'milestone-debt-free',      nameTh: 'ปลอดหนี้สมบูรณ์' },
    { id: 'milestone-profitable-bot', nameTh: 'บอทกำไรวันแรก' },
    { id: 'milestone-5050-allocation', nameTh: 'สมดุล 50/50 (เหรียญ vs บอท)' },
  ];

  const allMilestones: Milestone[] = ALL_MILESTONE_DEFS.map(({ id, nameTh }) => ({
    id: id as Milestone['id'],
    nameTh,
    unlocked:
      milestonesUnlocked.has(id) ||
      journal.milestones.some((m) => m.id === id),
  }));

  return (
    <div
      className="pwq-panel quest-journal-panel"
      role="dialog"
      aria-modal="true"
      aria-label="สมุดเป้าหมาย"
    >
      {/* Header */}
      <div className="pwq-panel-header">
        <h2 className="pwq-panel-title pwq-finance-readable">
          📖 สมุดเป้าหมาย
        </h2>
        <button
          className="pwq-panel-close"
          onClick={() => setOpenModal(null)}
          aria-label="ปิดสมุดเป้าหมาย"
        >
          ✕
        </button>
      </div>

      {/* Day label */}
      <div className="quest-day-label pwq-finance-readable">
        วันที่ {journal.date}
      </div>

      {/* First-run empty state */}
      {!anyCompleted && (
        <div className="quest-empty-state pwq-finance-readable">
          💡 ลองเริ่มต้นที่ฟาร์ม!
        </div>
      )}

      {/* Section: เป้าหมายวันนี้ */}
      <section aria-label="เป้าหมายวันนี้">
        <h3 className="quest-section-title pwq-finance-readable">
          เป้าหมายวันนี้
        </h3>
        <div className="quest-list">
          {journal.quests.map((quest) => (
            <QuestRow key={quest.id} quest={quest} />
          ))}
        </div>
      </section>

      {/* Section: เหรียญรางวัล (milestones) */}
      <section aria-label="เหรียญรางวัล">
        <h3 className="quest-section-title pwq-finance-readable">
          เหรียญรางวัล
        </h3>
        <div className="milestone-grid">
          {allMilestones.map((m) => (
            <MilestoneBadge key={m.id} milestone={m} />
          ))}
        </div>
      </section>
    </div>
  );
}
```

## Design Notes

- `buildDailyJournal` is called directly inside the component on every render — it is pure and
  cheap (~0.1ms), so no memoization is needed at this stage.
- `QuestJournalPanel` returns `null` when `openModal !== 'quest-journal'` — no portals or
  conditional wrappers needed in the parent.
- Milestone display in the panel shows **all 4 milestones** at all times (locked = greyed badge,
  unlocked = gold badge). This differs from `checkMilestones()` which returns only *newly* unlocked
  ones — the panel reconstructs full display state from `milestonesUnlocked` Set + newly unlocked.
- CSS variables (`--sim-lawn`, `--sim-parchment-dark`, etc.) must already be defined in the base
