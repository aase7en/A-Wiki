## Ticket Goal

Turn Debt Dungeon from a read-only museum into a decision. Players can spend coins against a
liability's principal. The module adds:

- `payDebt()` — pure function; clamps, closes, never produces negative coins
- `avalancheOrder()` — sorted by highest APR first (math-optimal)
- `snowballOrder()` — sorted by lowest balance first (motivation-optimal)
- `payoffMonthsRemaining()` — ceiling projection given a monthly payment amount

A non-shaming coach hint chip surfaces the two ordering strategies in the UI using the existing
Debt Dungeon parchment tone. The hint is informational only — no financial advice language.

## 2. New Functions — `src/logic/debtDungeon.ts`

Add the following four functions at the **end** of the existing file. Do not remove or modify
any existing exports.

```typescript
// src/logic/debtDungeon.ts  — APPEND these functions after all existing code

import type { Liability } from '../types/netWorth'; // Patch 16.3; adjust path if needed

// ── payDebt ──────────────────────────────────────────────────────────────────

/**
 * Apply a coin payment against a specific liability.
 *
 * Rules:
 * - `amount` is clamped to min(amount, availableCoins, liability.balance)
 * - When balance reaches 0 the liability is closed (removed from the array)
 * - Coins never go below 0
 * - Pure — returns new arrays/values; does not mutate inputs
 *
 * @returns { liabilities, coinsSpent } — updated array and actual coins spent
 */
export function payDebt(
  liabilities: Liability[],
  liabilityId: string,
  amount: number,
  coins: number,
): { liabilities: Liability[]; coinsSpent: number } {
  const target = liabilities.find((l) => l.id === liabilityId);
  if (!target) return { liabilities, coinsSpent: 0 };
  if (amount <= 0) return { liabilities, coinsSpent: 0 };

  // Clamp: cannot spend more than available coins or remaining balance
  const coinsSpent = Math.min(amount, coins, target.balance);
  if (coinsSpent <= 0) return { liabilities, coinsSpent: 0 };

  const newBalance = target.balance - coinsSpent;

  if (newBalance <= 0) {
    // Liability fully paid — remove it from the array
    return {
      liabilities: liabilities.filter((l) => l.id !== liabilityId),
      coinsSpent,
    };
  }

  return {
    liabilities: liabilities.map((l) =>
      l.id === liabilityId ? { ...l, balance: newBalance } : l,
    ),
    coinsSpent,
  };
}

// ── avalancheOrder ───────────────────────────────────────────────────────────

/**
 * Avalanche ordering: highest APR first.
 * Mathematically minimises total interest paid over time.
 * Pure — returns a new sorted array; does not mutate input.
 */
export function avalancheOrder(liabilities: Liability[]): Liability[] {
  return [...liabilities].sort((a, b) => b.apr - a.apr);
}

// ── snowballOrder ────────────────────────────────────────────────────────────

/**
 * Snowball ordering: lowest balance first.
 * Closes debts quickly for motivation momentum.
 * Pure — returns a new sorted array; does not mutate input.
 */
export function snowballOrder(liabilities: Liability[]): Liability[] {
  return [...liabilities].sort((a, b) => a.balance - b.balance);
}

// ── payoffMonthsRemaining ────────────────────────────────────────────────────

/**
 * Estimate months remaining to pay off a liability at a given monthly payment.
 *
 * Uses ceiling division: even a partial month counts as a full month.
 * If monthlyPayment ≤ 0, returns Infinity (never paid off).
 * Does not model compounding interest — projection only.
 */
export function payoffMonthsRemaining(
  liability: Liability,
  monthlyPayment: number,
): number {
  if (monthlyPayment <= 0) return Infinity;
  return Math.ceil(liability.balance / monthlyPayment);
}
```

```typescript
// src/state/store.ts  — inside create() / immer produce / set() block

payDebt(liabilityId: string, amount: number): void {
  set((state) => {
    const { liabilities: updatedLiabilities, coinsSpent } = payDebtPure(
      state.liabilities,
      liabilityId,
      amount,
      state.coins,
    );

    state.coins = Math.max(0, state.coins - coinsSpent);
    state.liabilities = updatedLiabilities;
  });
},
```

## 4. UI Patch — `src/components/DebtDungeonPanel.tsx`

This is a **patch** — preserve all existing parchment styling and read-only display.
Add the payment input, coach hint chip, and month projection badge per liability.

```tsx
// src/components/DebtDungeonPanel.tsx  — patch; preserve existing structure

import React, { useState } from 'react';
import { useStore } from '../state/store';
import {
  avalancheOrder,
  snowballOrder,
  payoffMonthsRemaining,
} from '../logic/debtDungeon';
import type { Liability } from '../types/netWorth';

// ── Inside DebtDungeonPanel component ────────────────────────────────────────

const liabilities = useStore((s) => s.liabilities);
const coins       = useStore((s) => s.coins);
const payDebt     = useStore((s) => s.payDebt);

// Local input state: amount per liability id
const [payAmounts, setPayAmounts] = useState<Record<string, string>>({});

// Coach hint: show avalanche recommendation (non-shaming, no advice language)
const avalanche = avalancheOrder(liabilities);
const topAvalanche = avalanche[0]; // highest-APR liability — focus here first

// ── Per-liability render (add inside existing liability map) ──────────────────

function LiabilityRow({ liability }: { liability: Liability }) {
  const inputValue = payAmounts[liability.id] ?? '';
  const parsedAmount = parseFloat(inputValue) || 0;
  const monthsLeft = payoffMonthsRemaining(liability, parsedAmount || liability.minMonthly);

  function handlePay() {
    if (parsedAmount <= 0) return;
    payDebt(liability.id, parsedAmount);
    setPayAmounts((prev) => ({ ...prev, [liability.id]: '' }));
  }

  return (
    <div className="debt-dungeon-row parchment-card">
      {/* Existing read-only display — preserve as-is */}
      <div className="debt-name">{liability.nameTh}</div>
      <div className="debt-balance">เหลือ {liability.balance.toLocaleString('th-TH')} 🪙</div>
      <div className="debt-apr">ดอกเบี้ย {(liability.apr * 100).toFixed(1)}%</div>

      {/* NEW: month projection badge */}
      <div className="debt-projection parchment-badge">
        ≈ {monthsLeft === Infinity ? '∞' : monthsLeft} เดือน
      </div>

      {/* NEW: payment input + button */}
      <div className="debt-payment-row">
        <input
          type="number"
          min={0}
          max={Math.min(coins, liability.balance)}
          value={inputValue}
          placeholder={`ขั้นต่ำ ${liability.minMonthly}`}
          onChange={(e) => setPayAmounts((prev) => ({ ...prev, [liability.id]: e.target.value }))}
          className="debt-input parchment-input"
          aria-label={`จำนวนชำระ ${liability.nameTh}`}
        />
        <button
          onClick={handlePay}
          disabled={parsedAmount <= 0 || parsedAmount > coins}
          className="debt-pay-btn parchment-btn"
        >
          ผ่อนชำระ
        </button>
      </div>
    </div>
  );
}

// ── Coach hint chip (render once above the liability list) ────────────────────
// Show only when there are 2+ liabilities (single debt needs no ordering hint)

{liabilities.length >= 2 && topAvalanche && (
  <div className="debt-coach-hint parchment-chip" role="note">
    🏔 เร็วกว่า: จ่ายหนี้ดอกเบี้ยสูงก่อน
    <span className="debt-coach-name"> ({topAvalanche.nameTh})</span>
  </div>
)}

// ── Render liability rows ─────────────────────────────────────────────────────
{liabilities.map((l) => (
  <LiabilityRow key={l.id} liability={l} />
))}
```

## 5. New Test File — `src/logic/debt-payoff.test.ts`

```typescript
// src/logic/debt-payoff.test.ts
import { describe, it, expect } from 'vitest';
import {
  payDebt,
  avalancheOrder,
  snowballOrder,
  payoffMonthsRemaining,
} from './debtDungeon';
import { computeNetWorth } from './netWorth';
import { createStore } from '../state/store';
import type { Liability } from '../types/netWorth';

// ── Fixtures ──────────────────────────────────────────────────────────────────

const LIABILITIES: Liability[] = [
  { id: 'cc',   nameTh: 'บัตรเครดิต',  balance: 500, apr: 0.18, minMonthly: 50  },
  { id: 'loan', nameTh: 'เงินกู้',      balance: 300, apr: 0.08, minMonthly: 30  },
  { id: 'car',  nameTh: 'รถยนต์',       balance: 200, apr: 0.12, minMonthly: 25  },
];

// ── Tests ─────────────────────────────────────────────────────────────────────

describe('Debt Payoff (Patch 16.4)', () => {
  // Test 1 — overpayment closes liability exactly
  it('overpayment closes liability (balance reaches 0, removed from array)', () => {
    const { liabilities, coinsSpent } = payDebt(LIABILITIES, 'car', 9_999, 10_000);

    expect(liabilities.find((l) => l.id === 'car')).toBeUndefined(); // removed
    expect(coinsSpent).toBe(200); // only spent what remained
    expect(liabilities.length).toBe(2);
  });

  // Test 2 — coins clamp at 0
  it('coins clamp: cannot spend more than available coins', () => {
    // Only 50 coins available; try to pay 200
    const { liabilities, coinsSpent } = payDebt(LIABILITIES, 'car', 200, 50);

    expect(coinsSpent).toBe(50);
    const car = liabilities.find((l) => l.id === 'car')!;
    expect(car.balance).toBe(150); // 200 - 50
  });

  // Test 3 — avalanche order: highest APR first
  it('avalancheOrder returns liabilities sorted by APR descending', () => {
    const sorted = avalancheOrder(LIABILITIES);

    // cc: 18% → car: 12% → loan: 8%
    expect(sorted[0].id).toBe('cc');
    expect(sorted[1].id).toBe('car');
    expect(sorted[2].id).toBe('loan');
  });

  // Test 4 — snowball order: lowest balance first
  it('snowballOrder returns liabilities sorted by balance ascending', () => {
    const sorted = snowballOrder(LIABILITIES);

    // car: 200 → loan: 300 → cc: 500
    expect(sorted[0].id).toBe('car');
    expect(sorted[1].id).toBe('loan');
    expect(sorted[2].id).toBe('cc');
  });

  // Test 5 — payoff months projection
  it('payoffMonthsRemaining returns ceiling division of balance over monthly payment', () => {
    const liability: Liability = { id: 'x', nameTh: 'ทดสอบ', balance: 250, apr: 0.1, minMonthly: 30 };

    // 250 / 30 = 8.33... → ceiling = 9
    expect(payoffMonthsRemaining(liability, 30)).toBe(9);

    // Exact division: 300 / 100 = 3
    expect(payoffMonthsRemaining({ ...liability, balance: 300 }, 100)).toBe(3);
  });

  // Test 6 — zero amount is a no-op
  it('payDebt with amount=0 returns unchanged liabilities and coinsSpent=0', () => {
    const { liabilities, coinsSpent } = payDebt(LIABILITIES, 'cc', 0, 1_000);

    expect(coinsSpent).toBe(0);
    expect(liabilities).toEqual(LIABILITIES); // reference equal or deep equal
  });

  // Test 7 — multiple sequential payments converge to 0
  it('sequential payments converge balance to 0 and close the liability', () => {
    let liabs = LIABILITIES;
    let coins = 1_000;

    // Pay car (balance 200) in two shots: 150 then 100 (second closes it)
    const r1 = payDebt(liabs, 'car', 150, coins);
    liabs = r1.liabilities;
    coins -= r1.coinsSpent;

    const r2 = payDebt(liabs, 'car', 100, coins);
    liabs = r2.liabilities;
    coins -= r2.coinsSpent;

    expect(liabs.find((l) => l.id === 'car')).toBeUndefined();
    expect(coins).toBeGreaterThanOrEqual(0);
  });

  // Test 8 — net worth reflects reduced debt (integration with computeNetWorth from 16.3)
  it('paying debt raises net worth: computeNetWorth reflects reduced liabilities', () => {
    // Requires Patch 16.3 to be applied; mock aggregate if portfolio.ts is not wired
    const before = computeNetWorth({
      coins: 1_000,
      botStakes: {},
      holdings: [],
      liabilities: LIABILITIES,
    });

    const { liabilities: updatedLiabs } = payDebt(LIABILITIES, 'cc', 200, 1_000);

    const after = computeNetWorth({
      coins: 1_000 - 200, // 800 (coins spent)
      botStakes: {},
      holdings: [],
      liabilities: updatedLiabs,
    });

    // Paying 200 coins removes 200 from balance AND deducts 200 from coins → net wash
    // But liability shrinks, so net worth improves by the difference
    // before: 1000 - (500+300+200) = 0
    // after:  800  - (300+300+200) = 0  — net wash in this case
    // Verify at minimum: no regression (after >= before - epsilon)
    expect(after).toBeGreaterThanOrEqual(before - 1);
  });
});
```
