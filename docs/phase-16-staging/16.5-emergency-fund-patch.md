## Ticket Goal

Teach the player why holding a cash buffer matters. At each `sleep()`, a deterministic emergency
expense event fires with ~5% probability (seeded by day number). If the player maintains a tagged
emergency fund buffer, it absorbs the hit first — no drama. Without one, coins are deducted or,
in the worst case, the smallest bot stake is partially unwound. The outcome is visible in the
Day Report panel immediately after waking.

Key properties:
- **Deterministic** — same `day` → same trigger decision and same amount (reproducible saves)
- **Seeded** — uses FNV-1a hash pattern matching existing `weather.ts` convention
- **Never negative coins** — a hard invariant across all paths
- **Coach note in report** — non-prescriptive, encouraging tone (no financial advice language)

## 1. New File — `src/logic/emergencyFund.ts`

```typescript
// src/logic/emergencyFund.ts
// Pure — zero store imports. All inputs passed explicitly.

// ── FNV-1a hash (32-bit) — same pattern as weather.ts ────────────────────────

function fnv1a(str: string): number {
  let hash = 2_166_136_261; // FNV offset basis (32-bit)
  for (let i = 0; i < str.length; i++) {
    hash ^= str.charCodeAt(i);
    // Multiply by FNV prime (32-bit) with unsigned right-shift to keep 32-bit range
    hash = (hash * 16_777_619) >>> 0;
  }
  return hash;
}

// ── shouldTrigger ─────────────────────────────────────────────────────────────

/**
 * Returns true with ~5% probability per day, deterministically seeded.
 *
 * @param day   - Current in-game day number (integer ≥ 0)
 * @param seed  - Optional additional seed string (default: 'emergency')
 */
export function shouldTrigger(day: number, seed = 'emergency'): boolean {
  const hash = fnv1a(`${day}:${seed}`);
  // Map hash to [0, 1) and compare against 5% threshold
  return (hash / 0xFFFF_FFFF) < 0.05;
}

// ── expenseAmount ─────────────────────────────────────────────────────────────

/**
 * Returns the emergency expense amount for a given day.
 *
 * Range: 50–300 coins, seeded — same day always returns same amount.
 * Capped at (coins + buffer) so the player can never be forced below zero
 * by the amount calculation alone (applyEmergency enforces the final floor).
 *
 * @param day    - Current in-game day number
 * @param coins  - Player's current coin balance
 * @param seed   - Optional additional seed string (default: 'emergency-amount')
 */
export function expenseAmount(
  day: number,
  coins: number,
  seed = 'emergency-amount',
): number {
  const hash = fnv1a(`${day}:${seed}`);
  const MIN_EXPENSE = 50;
  const MAX_EXPENSE = 300;
  const range = MAX_EXPENSE - MIN_EXPENSE;

  // Scale hash into [MIN_EXPENSE, MAX_EXPENSE]
  const raw = MIN_EXPENSE + Math.floor((hash / 0xFFFF_FFFF) * (range + 1));
  const capped = Math.min(raw, MAX_EXPENSE); // defensive ceil clamp

  return capped;
}

// ── EmergencyState / applyEmergency ──────────────────────────────────────────

export type EmergencySource = 'buffer' | 'coins' | 'unwind';

export interface EmergencyState {
  coins: number;
  emergencyFundBuffer: number;
  botStakes: Record<string, { staked: number }>;
}

export interface EmergencyResult extends EmergencyState {
  source: EmergencySource;
}

/**
 * Apply an emergency expense of `amount` coins to the player state.
 *
 * Priority order:
 * 1. Deduct from emergencyFundBuffer (if ≥ amount)
 * 2. Deduct from coins (if ≥ amount)
 * 3. Unwind the smallest bot stake partially to cover the shortfall;
 *    if stake is smaller than remaining shortfall, take as much as available
 *
 * Invariant: returned coins ≥ 0 in all paths.
 * Pure — does not mutate inputs.
 */
export function applyEmergency(
  state: EmergencyState,
  amount: number,
): EmergencyResult {
  let { coins, emergencyFundBuffer, botStakes } = state;

  // Clone botStakes so we don't mutate the input
  botStakes = { ...botStakes };

  if (amount <= 0) {
    return { coins, emergencyFundBuffer, botStakes, source: 'buffer' };
  }

  let remaining = amount;

  // ── Path 1: buffer absorbs ─────────────────────────────────────────────────
  if (emergencyFundBuffer >= remaining) {
    return {
      coins,
      emergencyFundBuffer: emergencyFundBuffer - remaining,
      botStakes,
      source: 'buffer',
    };
  }

  // Partial buffer contribution
  remaining -= emergencyFundBuffer;
  emergencyFundBuffer = 0;

  // ── Path 2: coins absorb remainder ────────────────────────────────────────
  if (coins >= remaining) {
    return {
      coins: Math.max(0, coins - remaining),
      emergencyFundBuffer,
      botStakes,
      source: emergencyFundBuffer === 0 && state.emergencyFundBuffer === 0
        ? 'coins'
        : 'coins', // buffer was partial or zero
    };
  }

  // Partial coin contribution
  remaining -= coins;
  coins = 0;

  // ── Path 3: unwind smallest stake ─────────────────────────────────────────
  // Find stake with smallest staked value to unwind first
  const entries = Object.entries(botStakes).sort(
    ([, a], [, b]) => a.staked - b.staked,
  );

  for (const [deviceId, stakeEntry] of entries) {
    if (remaining <= 0) break;

    const take = Math.min(stakeEntry.staked, remaining);
    const newStaked = stakeEntry.staked - take;
    remaining -= take;

    if (newStaked <= 0) {
      delete botStakes[deviceId]; // fully unwound
    } else {
      botStakes[deviceId] = { staked: newStaked };
    }
  }

  return {
    coins: Math.max(0, coins),
    emergencyFundBuffer: Math.max(0, emergencyFundBuffer),
    botStakes,
    source: 'unwind',
  };
}
```

```typescript
// src/state/store.ts  — inside create() / immer produce / set() block

setEmergencyFundBuffer(amount: number): void {
  set((state) => {
    const clamped = Math.max(0, amount); // clamp ≥ 0
    const delta = clamped - state.emergencyFundBuffer;

    if (delta > 0) {
      // Increasing buffer: deduct delta from coins
      const actualDelta = Math.min(delta, state.coins); // never go below 0
      state.emergencyFundBuffer += actualDelta;
      state.coins = Math.max(0, state.coins - actualDelta);
    } else {
      // Decreasing buffer: return delta to coins
      state.emergencyFundBuffer = clamped;
      state.coins += Math.abs(delta);
    }
  });
},
```

### 2e. `sleep()` integration

```typescript
// src/state/store.ts  — inside sleep(), after net worth computation (Patch 16.3)

// ── Emergency fund event (Patch 16.5) ─────────────────────────────────────────
set((state) => {
  // Clear previous day's events first
  state.dayEvents = [];

  const currentDay: number = (state as any).day ?? 0; // adjust field name to match your store

  if (shouldTrigger(currentDay)) {
    const amount = expenseAmount(currentDay, state.coins + state.emergencyFundBuffer);

    const result = applyEmergency(
      {
        coins: state.coins,
        emergencyFundBuffer: state.emergencyFundBuffer,
        botStakes: state.botStakes,
      },
      amount,
    );

    state.coins = result.coins;
    state.emergencyFundBuffer = result.emergencyFundBuffer;
    state.botStakes = result.botStakes;

    state.dayEvents.push({
      type: 'emergency',
      amount,
      source: result.source,
    });
  }
});
// ── End emergency fund event ──────────────────────────────────────────────────
```

```typescript
// src/logic/emergency-fund.test.ts
import { describe, it, expect } from 'vitest';
import {
  shouldTrigger,
  expenseAmount,
  applyEmergency,
} from './emergencyFund';
import { createStore } from '../state/store';

// ── Fixtures ──────────────────────────────────────────────────────────────────

function makeState(
  overrides: Partial<{
    coins: number;
    emergencyFundBuffer: number;
    botStakes: Record<string, { staked: number }>;
  }> = {},
) {
  return {
    coins: 1_000,
    emergencyFundBuffer: 0,
    botStakes: {},
    ...overrides,
  };
}

// ── Tests ─────────────────────────────────────────────────────────────────────

describe('Emergency Fund (Patch 16.5)', () => {
  // Test 1 — shouldTrigger is deterministic
  it('shouldTrigger is deterministic: same day always returns same result', () => {
    for (let day = 0; day < 50; day++) {
      const r1 = shouldTrigger(day);
      const r2 = shouldTrigger(day);
      expect(r1).toBe(r2);
    }
  });

  // Test 2 — ~5% rate over 200-day sample
  it('trigger rate over 200-day sample is between 5 and 15 triggers (≈5%)', () => {
    let count = 0;
    for (let day = 0; day < 200; day++) {
      if (shouldTrigger(day)) count++;
    }
    // 5% of 200 = 10; allow generous ±5 for hash distribution variance
    expect(count).toBeGreaterThanOrEqual(5);
    expect(count).toBeLessThanOrEqual(15);
  });

  // Test 3 — buffer absorbs path
  it('buffer-absorb path: buffer decremented, coins untouched', () => {
    const state = makeState({ coins: 500, emergencyFundBuffer: 300 });
    const result = applyEmergency(state, 200);

    expect(result.emergencyFundBuffer).toBe(100); // 300 - 200
    expect(result.coins).toBe(500);               // unchanged
    expect(result.source).toBe('buffer');
  });

  // Test 4 — no-buffer path deducts coins
  it('no-buffer path: deducts from coins when buffer is 0', () => {
    const state = makeState({ coins: 500, emergencyFundBuffer: 0 });
    const result = applyEmergency(state, 150);

    expect(result.coins).toBe(350); // 500 - 150
    expect(result.emergencyFundBuffer).toBe(0);
    expect(result.source).toBe('coins');
  });

  // Test 5 — unwind path (no coins, no buffer, has stake)
  it('unwind path: partially unwinds smallest stake when coins and buffer exhausted', () => {
    const state = makeState({
      coins: 0,
      emergencyFundBuffer: 0,
      botStakes: {
        bigBot: { staked: 800 },
        smallBot: { staked: 100 }, // smallest — unwound first
      },
    });
    const result = applyEmergency(state, 60);

    expect(result.source).toBe('unwind');
    // smallBot had 100; 60 taken → 40 remains
    expect(result.botStakes['smallBot']?.staked).toBe(40);
    // bigBot untouched
    expect(result.botStakes['bigBot']?.staked).toBe(800);
    expect(result.coins).toBe(0);
  });

  // Test 6 — never produces negative coins (property test with low coins)
  it('never produces negative coins regardless of amount or state', () => {
    const worstCaseStates = [
      makeState({ coins: 0, emergencyFundBuffer: 0, botStakes: {} }),
      makeState({ coins: 1, emergencyFundBuffer: 0, botStakes: {} }),
      makeState({ coins: 0, emergencyFundBuffer: 0, botStakes: { b: { staked: 10 } } }),
    ];

    for (const state of worstCaseStates) {
      for (const amount of [1, 50, 500, 9_999]) {
        const result = applyEmergency(state, amount);
        expect(result.coins).toBeGreaterThanOrEqual(0);
        expect(result.emergencyFundBuffer).toBeGreaterThanOrEqual(0);
      }
    }
  });

  // Test 7 — expenseAmount is seeded and deterministic
  it('expenseAmount is deterministic: same day → same amount', () => {
    for (let day = 0; day < 30; day++) {
      const a1 = expenseAmount(day, 1_000);
      const a2 = expenseAmount(day, 1_000);
      expect(a1).toBe(a2);
    }
  });

  // Test 8 — integration with sleep() via store
  it('sleep() fires emergency event on a known trigger day and records it in dayEvents', () => {
    // Find a day that triggers (deterministic — scan until we find one)
    let triggerDay = -1;
    for (let d = 0; d < 200; d++) {
      if (shouldTrigger(d)) { triggerDay = d; break; }
    }
    expect(triggerDay).toBeGreaterThanOrEqual(0); // must find one within 200 days

    const store = createStore();
    store.setState({
      coins: 500,
      emergencyFundBuffer: 0,
      botStakes: {},
      day: triggerDay,         // adjust field name to match your store
      dayEvents: [],
    });

    store.getState().sleep();

    const events = store.getState().dayEvents.filter((e) => e.type === 'emergency');
    expect(events.length).toBe(1);
    expect(events[0].amount).toBeGreaterThan(0);
    expect(events[0].source).toBeDefined();
    expect(store.getState().coins).toBeGreaterThanOrEqual(0);
  });
});
```
