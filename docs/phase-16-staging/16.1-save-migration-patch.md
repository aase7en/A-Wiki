## Problem

`parseSave` at `src/logic/saveGame.ts:166` currently does:

```ts
if (parsed.version !== SAVE_VERSION) return null;
```

This hard-rejects any save whose version doesn't exactly match `SAVE_VERSION`. When Phase 16 bumps the version (to add `botStakes`, `netWorthHistory`, `emergencyFundBuffer`, `debtPayments`), every existing v2 player gets their save wiped on first load. The fix: replace with a `migrate(fromVersion → SAVE_VERSION)` ladder.

---

## Changes to `src/logic/saveGame.ts`

### 1. Add migration ladder (insert before `parseSave`)

```typescript
// ---------------------------------------------------------------------------
// Migration ladder
// Each step migrates raw save data from version N to N+1.
// Rules:
//   - Keep steps small and additive — only set defaults for fields added in that bump.
//   - Never remove or rename fields here; do that in the target-version step only.
//   - Steps MUST be pure (no side effects, no async, no store imports).
// ---------------------------------------------------------------------------
type RawSave = Record<string, unknown>;

/** Migrate a raw save from `fromVersion` up to `SAVE_VERSION`. Returns null if
 *  the save is from a newer version (downgrade not supported). */
function migrateRaw(raw: RawSave, fromVersion: number): RawSave | null {
  if (fromVersion > SAVE_VERSION) {
    // Save created by a newer game build — fail closed, start fresh
    return null;
  }

  let data: RawSave = { ...raw };

  if (fromVersion < 2) {
    // v1 → v2 (pre-Phase7 save): synthesize minimum viable v2 shape.
    // In practice these saves predate versioning, so all fields get defaults.
    data = {
      coins: typeof data.coins === 'number' ? data.coins : 100,
      day: typeof data.day === 'number' ? data.day : 1,
      ...data,
      version: 2,
    };
    fromVersion = 2;
  }

  // ── Phase 16 migration steps — each ticket adds its step here ──────────
  // Ticket 16.2 will add:
  // if (fromVersion < 3) {
  //   data = { ...data, botStakes: {}, version: 3 };
  //   fromVersion = 3;
  // }
  // Ticket 16.3 will add:
  // if (fromVersion < 4) {
  //   data = { ...data, netWorthHistory: [], version: 4 };
  //   fromVersion = 4;
  // }
  // Ticket 16.4 will add:
  // if (fromVersion < 5) {
  //   data = { ...data, debtPayments: {}, version: 5 };
  //   fromVersion = 5;
  // }
  // Ticket 16.5 will add:
  // if (fromVersion < 6) {
  //   data = { ...data, emergencyFundBuffer: 0, version: 6 };
  //   fromVersion = 6;
  // }
  // ───────────────────────────────────────────────────────────────────────

  return data;
}
```

### 2. Replace the hard-reject at line ~166

**Before:**
```typescript
if (parsed.version !== SAVE_VERSION) return null;
```

**After:**
```typescript
const rawVersion = typeof parsed.version === 'number' ? parsed.version : -1;
if (rawVersion < 0) return null; // corrupt/missing version field → fresh game

const migrated = migrateRaw(parsed as RawSave, rawVersion);
if (migrated === null) return null; // downgrade not supported → fresh game

// Replace `parsed` with the migrated data for the rest of parseSave
const safe = migrated;
// (continue using `safe` instead of `parsed` for all field extractions below)
```

> **Note**: After the swap, any references to `parsed.fieldName` below line 166 must change to `safe.fieldName`. Grep for `parsed\.` within `parseSave` and update each one.

## New tests for `src/logic/saveGame.test.ts`

Add the following test suite (test-first: write these before applying the patch above):

```typescript
describe('parseSave migration ladder', () => {
  const CURRENT = SAVE_VERSION; // imported from saveGame.ts

  function makeV2Save(overrides: Record<string, unknown> = {}) {
    // Minimal valid v2 save — the shape that exists on disk right now
    return JSON.stringify({
      version: 2,
      coins: 500,
      day: 14,
      inventory: {},
      plots: [],
      iotSlots: [],
      botConfigs: {},
      // ... other v2 fields your parseSave expects
      ...overrides,
    });
  }

  it('loads a v2 save with missing Phase-16 fields using sensible defaults (no wipe)', () => {
    // v2 save has no botStakes/netWorthHistory/etc — must not return null
    const result = parseSave(makeV2Save());
    expect(result).not.toBeNull();
    expect(result?.coins).toBe(500);
    expect(result?.day).toBe(14);
  });

  it('loads current-version save unchanged', () => {
    const blob = makeV2Save({ version: CURRENT });
    const result = parseSave(blob);
    expect(result).not.toBeNull();
  });

  it('returns null (fresh game) for a save from a newer version (downgrade)', () => {
    const blob = makeV2Save({ version: CURRENT + 10 });
    expect(parseSave(blob)).toBeNull();
  });

  it('returns null for a corrupt/unparseable blob', () => {
    expect(parseSave('not-json-at-all')).toBeNull();
    expect(parseSave('{broken')).toBeNull();
    expect(parseSave('null')).toBeNull();
  });

  it('returns null if version field is missing or non-numeric', () => {
    expect(parseSave(JSON.stringify({ coins: 100 }))).toBeNull();
    expect(parseSave(JSON.stringify({ version: 'two', coins: 100 }))).toBeNull();
  });

  it('migration is deterministic — same input always yields same output', () => {
    const blob = makeV2Save();
    const a = parseSave(blob);
    const b = parseSave(blob);
    expect(a).toEqual(b);
  });

  it('v1 save (no version field assumed → -1) fails closed to fresh game', () => {
    // v1 pre-Phase7 saves won't have a `version` key
    const v1Blob = JSON.stringify({ coins: 50, day: 3 });
    // With no version field, rawVersion = -1 → return null
    expect(parseSave(v1Blob)).toBeNull();
  });
});
```
