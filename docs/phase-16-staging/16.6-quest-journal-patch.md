```typescript
// src/logic/questJournal.ts
// Ticket 16.6 — Daily Quest Journal (pure logic, no React/Phaser)
// Depends on: 16.3 netWorth, 16.1 botStakes/coins/liabilities in store snapshot

// ---------------------------------------------------------------------------
// Types

/** All quest IDs — add new ones here and to QUEST_META below */
export type QuestId =

  | 'town-npc'

  | 'analyst-desk';

export interface Quest {
  id: QuestId;
  /** Thai display name */
  nameTh: string;
  /** Thai description / instruction */
  descTh: string;
  /** Current progress count (0–goal) */
  progress: number;
  /** Target count to complete */
  goal: number;
  /** true when progress >= goal */
  completed: boolean;
  /** Short Thai hint pointing to the next action */
  nextActionHint: string;
}

export type MilestoneId =
  | 'milestone-10k-networth'
  | 'milestone-debt-free'
  | 'milestone-profitable-bot'
  | 'milestone-5050-allocation';

export interface Milestone {
  id: MilestoneId;
  /** Thai display name */
  nameTh: string;
  /** Whether this milestone has just been newly unlocked in this check */
  unlocked: boolean;
}

export interface DailyJournal {
  /** Game day number */
  date: number;
  /** Active quests for this day */
  quests: Quest[];
  /** Newly unlocked milestones (already-unlocked ones are excluded) */
  milestones: Milestone[];
}

/** Subset of a liability record needed by journal logic */
export interface Liability {
  id: string;
  balance: number;
}

/**
 * Snapshot of store state that questJournal.ts needs.
 * Pass this in from the Zustand selector — never import store directly.
 */
export interface JournalInputs {
  /** Current coin balance (economy) */
  coins: number;
  /** Computed net worth from 16.3 */
  netWorth: number;
  /** Historical net worth per day (index = day − 1) */
  netWorthHistory: number[];
  /** Bot positions: Record<ticker, { staked: number }> */
  botStakes: Record<string, { staked: number }>;
  /** All active liabilities from the debt system (16.4) */
  liabilities: Liability[];
  /** PnL of the most recent bot settlement cycle (+ = profit, − = loss, 0 = no cycle yet) */
  lastBotPnl: number;
  /** How many plots the player has harvested today */
  plotsHarvested: number;
  /** How many plots the player has watered today */
  plotsWatered: number;
  /** How many plots the player has seeded today */
  plotsSeeded: number;
  /** Coins earned from shipping produce today */
  coinsShipped: number;
  /** True if all animals were fed today */
  animalsFed: boolean;
  /** True if at least one NPC was talked to today */
  npcTalkedToday: boolean;
  /** True if the analyst desk was opened at least once today */
  analystDeskOpened: boolean;
  /** True if today is a festival day (from calendar.ts) */
  isFestivalDay: boolean;
  /** Set of milestone IDs already permanently unlocked (persisted in store) */
  milestonesUnlocked: Set<string>;
}

// Milestone definitions
// ---------------------------------------------------------------------------

const ALL_MILESTONES: Array<{ id: MilestoneId; nameTh: string }> = [
  { id: 'milestone-10k-networth', nameTh: 'มูลค่าสุทธิ ฿10,000 แรก' },
  { id: 'milestone-debt-free', nameTh: 'ปลอดหนี้สมบูรณ์' },
  { id: 'milestone-profitable-bot', nameTh: 'บอทกำไรวันแรก' },
  { id: 'milestone-5050-allocation', nameTh: 'สมดุล 50/50 (เหรียญ vs บอท)' },
];

// ---------------------------------------------------------------------------
// Internal helpers
// ---------------------------------------------------------------------------

function sumBotStakes(botStakes: Record<string, { staked: number }>): number {
  return Object.values(botStakes).reduce((acc, pos) => acc + pos.staked, 0);
}

function isMilestone10kUnlocked(inputs: JournalInputs): boolean {
  return inputs.netWorth >= 10_000;
}

function isMilestoneDebtFreeUnlocked(inputs: JournalInputs): boolean {
  return (
    inputs.liabilities.length === 0 ||
    inputs.liabilities.every((l) => l.balance <= 0)
  );
}

function isMilestoneProfitableBotUnlocked(inputs: JournalInputs): boolean {
  return inputs.lastBotPnl > 0;
}

function isMilestone5050Unlocked(inputs: JournalInputs): boolean {
  const botSum = sumBotStakes(inputs.botStakes);
  const total = inputs.coins + botSum;
  if (total <= 0) return false;
  const imbalance = Math.abs(botSum - inputs.coins) / total;
  return imbalance < 0.1; // within 10% of 50/50
}

const MILESTONE_CHECKS: Record<
  MilestoneId,
  (inputs: JournalInputs) => boolean
> = {
  'milestone-10k-networth': isMilestone10kUnlocked,
  'milestone-debt-free': isMilestoneDebtFreeUnlocked,
  'milestone-profitable-bot': isMilestoneProfitableBotUnlocked,
  'milestone-5050-allocation': isMilestone5050Unlocked,
};

// ---------------------------------------------------------------------------
// buildQuest — construct a single Quest from inputs
// ---------------------------------------------------------------------------

function buildQuest(id: QuestId, rawProgress: number): Quest {
  const meta = QUEST_META[id];
  const progress = Math.min(rawProgress, meta.goal);
  const completed = progress >= meta.goal;
  return {
    id,
    nameTh: meta.nameTh,
    descTh: meta.descTh,
    progress,
    goal: meta.goal,
    completed,
    nextActionHint: completed ? '✅ เสร็จแล้ว!' : meta.nextActionHint,
  };
}

// ---------------------------------------------------------------------------
// Public API
// ---------------------------------------------------------------------------

/**
 * Check which milestones are **newly** unlocked (not already in milestonesUnlocked).
 * Returns only milestones that cross from locked → unlocked in this snapshot.
 */
export function checkMilestones(inputs: JournalInputs): Milestone[] {
  return ALL_MILESTONES.filter(({ id }) => {
    if (inputs.milestonesUnlocked.has(id)) return false; // already awarded
    return MILESTONE_CHECKS[id](inputs);
  }).map(({ id, nameTh }) => ({ id, nameTh, unlocked: true }));
}

/**
 * Pure function: build the full DailyJournal for a given day from a state snapshot.
 *
 * - Always includes standard farm/town/animal/analyst quests.
 * - Festival quest only included on festival days.
 * - Milestone check filters out already-unlocked milestones.
 *
 * @param inputs   Snapshot of store state fields needed by journal logic.
 * @param day      Current game day number (1-indexed).
 */
export function buildDailyJournal(
  inputs: JournalInputs,
  day: number,
): DailyJournal {
  const quests: Quest[] = [
    buildQuest('farm-seed', inputs.plotsSeeded),
    buildQuest('farm-water', inputs.plotsWatered),
    buildQuest('farm-harvest', inputs.plotsHarvested),
    buildQuest('farm-ship', inputs.coinsShipped),
    buildQuest('town-market', inputs.npcTalkedToday ? 1 : 0), // market counts if in town
    buildQuest('town-npc', inputs.npcTalkedToday ? 1 : 0),
    buildQuest('town-buy', 0), // purchase tracking TBD by 16.8
    buildQuest('animal-feed', inputs.animalsFed ? 1 : 0),
    buildQuest('analyst-desk', inputs.analystDeskOpened ? 1 : 0),
  ];

  if (inputs.isFestivalDay) {
    quests.push(buildQuest('festival-attend', 0)); // progress updated separately via festival system
  }

  const milestones = checkMilestones(inputs);

  return { date: day, quests, milestones };
}
```

```typescript
// src/logic/questJournal.test.ts
// Ticket 16.6 — 10 Vitest tests for questJournal pure logic

import { describe, it, expect } from 'vitest';
import {
  buildDailyJournal,
  checkMilestones,
  type JournalInputs,
  type Liability,
} from './questJournal';

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

function makeInputs(overrides: Partial<JournalInputs> = {}): JournalInputs {
  return {
    coins: 500,
    netWorth: 500,
    netWorthHistory: [500],
    botStakes: {},
    liabilities: [],
    lastBotPnl: 0,
    plotsHarvested: 0,
    plotsWatered: 0,
    plotsSeeded: 0,
    coinsShipped: 0,
    animalsFed: false,
    npcTalkedToday: false,
    analystDeskOpened: false,
    isFestivalDay: false,
    milestonesUnlocked: new Set<string>(),
    ...overrides,
  };
}

// ---------------------------------------------------------------------------
// Tests
// ---------------------------------------------------------------------------

describe('buildDailyJournal', () => {
  it('1. Normal day: returns correct quests with correct progress', () => {
    const inputs = makeInputs({
      plotsSeeded: 2,
      plotsWatered: 3,
      plotsHarvested: 1,
      coinsShipped: 30,
      analystDeskOpened: true,
    });
    const journal = buildDailyJournal(inputs, 5);

    expect(journal.date).toBe(5);

    const seedQuest = journal.quests.find((q) => q.id === 'farm-seed')!;
    expect(seedQuest.progress).toBe(2);
    expect(seedQuest.goal).toBe(3);
    expect(seedQuest.completed).toBe(false);

    const waterQuest = journal.quests.find((q) => q.id === 'farm-water')!;
    expect(waterQuest.progress).toBe(3);
    expect(waterQuest.completed).toBe(true);

    const analystQuest = journal.quests.find((q) => q.id === 'analyst-desk')!;
    expect(analystQuest.progress).toBe(1);
    expect(analystQuest.completed).toBe(true);
  });

  it('2. Festival day: festival quest is included', () => {
    const inputs = makeInputs({ isFestivalDay: true });
    const journal = buildDailyJournal(inputs, 10);

    const festivalQuest = journal.quests.find((q) => q.id === 'festival-attend');
    expect(festivalQuest).toBeDefined();
  });

  it('3. Non-festival day: festival quest is NOT included', () => {
    const inputs = makeInputs({ isFestivalDay: false });
    const journal = buildDailyJournal(inputs, 11);

    const festivalQuest = journal.quests.find((q) => q.id === 'festival-attend');
    expect(festivalQuest).toBeUndefined();
  });

  it('4. Completed quest: progress equals goal and completed is true', () => {
    const inputs = makeInputs({
      plotsHarvested: 5, // goal is 2, so capped at 2 and completed
    });
    const journal = buildDailyJournal(inputs, 1);

    const harvestQuest = journal.quests.find((q) => q.id === 'farm-harvest')!;
    expect(harvestQuest.progress).toBe(harvestQuest.goal);
    expect(harvestQuest.completed).toBe(true);
    expect(harvestQuest.nextActionHint).toContain('✅');
  });

  it('5. Milestone ฿10k fires when netWorth >= 10000 and not yet unlocked', () => {
    const inputs = makeInputs({
      netWorth: 12_000,
      milestonesUnlocked: new Set<string>(), // not unlocked
    });
    const milestones = checkMilestones(inputs);

    const tenk = milestones.find((m) => m.id === 'milestone-10k-networth');
    expect(tenk).toBeDefined();
    expect(tenk!.unlocked).toBe(true);
  });

  it('6. Milestone debt-free fires when all liabilities have balance 0', () => {
    const liabilities: Liability[] = [
      { id: 'loan-1', balance: 0 },
      { id: 'loan-2', balance: 0 },
    ];
    const inputs = makeInputs({ liabilities });
    const milestones = checkMilestones(inputs);

    const debtFree = milestones.find((m) => m.id === 'milestone-debt-free');
    expect(debtFree).toBeDefined();
    expect(debtFree!.unlocked).toBe(true);
  });

  it('7. Milestone profitable bot fires when lastBotPnl > 0', () => {
    const inputs = makeInputs({ lastBotPnl: 42 });
    const milestones = checkMilestones(inputs);

    const profitBot = milestones.find(
      (m) => m.id === 'milestone-profitable-bot',
    );
    expect(profitBot).toBeDefined();
    expect(profitBot!.unlocked).toBe(true);
  });

  it('8. Milestone 50/50 fires when botStakesSum ≈ coins within 10%', () => {
    // coins = 1000, botStakes total = 1050 → imbalance = |1050-1000|/2050 ≈ 2.4% < 10%
    const inputs = makeInputs({
      coins: 1_000,
      botStakes: { TICKER_A: { staked: 500 }, TICKER_B: { staked: 550 } },
    });
    const milestones = checkMilestones(inputs);

    const fiftyfifty = milestones.find(
      (m) => m.id === 'milestone-5050-allocation',
    );
    expect(fiftyfifty).toBeDefined();
    expect(fiftyfifty!.unlocked).toBe(true);
  });

  it('9. Milestones do NOT fire if already in milestonesUnlocked set', () => {
    const inputs = makeInputs({
      netWorth: 50_000,
      lastBotPnl: 100,
      liabilities: [],
      coins: 1_000,
      botStakes: { A: { staked: 1_000 } },
      milestonesUnlocked: new Set<string>([
        'milestone-10k-networth',
        'milestone-debt-free',
        'milestone-profitable-bot',
        'milestone-5050-allocation',
      ]),
    });
    const milestones = checkMilestones(inputs);

    // All already unlocked — nothing should be returned
    expect(milestones).toHaveLength(0);
  });

  it('10. Fully completed day: all standard quests done, all milestones hit', () => {
    const inputs = makeInputs({
      coins: 1_000,
      netWorth: 15_000,
      botStakes: { A: { staked: 1_000 } },
      liabilities: [],
      lastBotPnl: 99,
      plotsSeeded: 10,
      plotsWatered: 10,
      plotsHarvested: 10,
      coinsShipped: 500,
      animalsFed: true,
      npcTalkedToday: true,
      analystDeskOpened: true,
      isFestivalDay: false,
      milestonesUnlocked: new Set<string>(),
    });
    const journal = buildDailyJournal(inputs, 42);

    // All quests completed
    const incomplete = journal.quests.filter((q) => !q.completed);
    // town-buy is not auto-completed (no purchase tracking yet) — exclude it
    const incompleteExcludingBuy = incomplete.filter((q) => q.id !== 'town-buy');
    expect(incompleteExcludingBuy).toHaveLength(0);

    // All 4 milestones newly unlocked
    expect(journal.milestones).toHaveLength(4);
    expect(journal.milestones.every((m) => m.unlocked)).toBe(true);
  });
});
```

## Design Notes

- `buildDailyJournal` is a pure function — safe to call on every render without side effects.
- `checkMilestones` only returns **newly unlocked** milestones; the caller owns the `milestonesUnlocked` Set and must persist it.
- `town-buy` quest progress is intentionally 0 until ticket 16.8 adds purchase tracking to the store.
- The festival quest has `progress: 0` by default; the festival system (from prior phases) should update this field by dispatching a store action that increments `festivalAttended` (add it to `JournalInputs` in 16.8 if needed).
- `farm-ship` uses `coinsShipped` (coins earned, not units shipped) to match the shipping loop from prior phases.
