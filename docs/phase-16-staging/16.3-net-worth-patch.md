## Ticket Goal

Replace the static seed number in the HUD with a live, computed net worth figure:

```
netWorth = coins + Σ botStakes[].staked + portfolio.aggregate(holdings).netWorth − Σ liabilities[].balance
```

Recomputed every `sleep()`. A rolling 28-day history array (one point per sleep) feeds a
30×16 px SVG sparkline beside the HUD number. The computation lives in a pure module so it is
trivially testable without any store dependency.

## 2. New Pure Module — `src/logic/netWorth.ts`

Zero store imports — all inputs passed explicitly. This keeps the function testable in isolation.

```typescript
// src/logic/netWorth.ts

import { aggregate } from './portfolio';
import type { NetWorthInputs } from '../types/netWorth';

/**
 * Compute player net worth at a point in time.
 *
 * Formula:
 *   netWorth = coins
 *            + Σ botStakes[].staked           (locked capital not yet settled)
 *            + portfolio.aggregate(holdings).netWorth
 *            - Σ liabilities[].balance
 *
 * Returns a number. Can be negative when liabilities exceed all assets.
 * Pure — deterministic given the same inputs. No side-effects.
 */
export function computeNetWorth(inputs: NetWorthInputs): number {
  const { coins, botStakes, holdings, liabilities } = inputs;

  const stakedTotal = Object.values(botStakes).reduce(
    (sum, entry) => sum + entry.staked,
    0,
  );

  // portfolio.aggregate() is allowed to return 0 for an empty holdings array
  const portfolioValue = holdings.length > 0
    ? aggregate(holdings).netWorth
    : 0;

  const liabilityTotal = liabilities.reduce(
    (sum, l) => sum + l.balance,
    0,
  );

  return coins + stakedTotal + portfolioValue - liabilityTotal;
}
```

```typescript
set((state) => {

    coins: state.coins,
    botStakes: state.botStakes,
    holdings: (state as any).holdings ?? [],   // adjust field name to match your store
    liabilities: (state as any).liabilities ?? [],
  };

  const nw = computeNetWorth(inputs);

  // Append new value; trim to last 28 entries (one per sleep = one per in-game day)
  const raw = [...state.netWorthHistory, nw];
  const trimmedHistory = raw.length > 28 ? raw.slice(raw.length - 28) : raw;

  return {
    netWorth: nw,
    netWorthHistory: trimmedHistory,
  };
});
```

## 4. HUD Component Patch — `src/components/HudOverlay.tsx`

Replace the static portfolio/net-worth number and add a 30×16 px SVG sparkline.
This is a **patch** — do not replace the entire file.

```tsx
// src/components/HudOverlay.tsx  — patch only; preserve all surrounding JSX

import { useStore } from '../state/store';

// ── Inside HudOverlay component — add selectors ───────────────────────────────
const netWorth = useStore((s) => s.netWorth);
const history  = useStore((s) => s.netWorthHistory);

// ── Sparkline sub-component (can live in same file or extracted) ──────────────

function NetWorthSparkline({ data }: { data: number[] }) {
  if (data.length < 2) return null;

  const W = 30;
  const H = 16;
  const min = Math.min(...data);
  const max = Math.max(...data);
  // Avoid division by zero when all values are equal (flat line)
  const range = max - min || 1;

  const points = data
    .map((v, i) => {
      const x = (i / (data.length - 1)) * W;
      const y = H - ((v - min) / range) * H;
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    })
    .join(' ');

  return (
    <svg
      width={W}
      height={H}
      viewBox={`0 0 ${W} ${H}`}
      aria-label="Net worth trend"
      style={{ display: 'inline-block', verticalAlign: 'middle', marginLeft: 4 }}
    >
      <polyline
        points={points}
        fill="none"
        stroke="var(--sim-lawn)"
        strokeWidth="1.5"
        strokeLinejoin="round"
        strokeLinecap="round"
      />
    </svg>
  );
}

// ── Replace static number with live value + sparkline ────────────────────────
// Find the existing static render (e.g. <span>1,234 🪙</span>) and replace with:

<span className="hud-net-worth">
  {netWorth.toLocaleString('th-TH', { maximumFractionDigits: 0 })} 🪙
  <NetWorthSparkline data={history} />
</span>
```

```typescript
// ── Fixtures ──────────────────────────────────────────────────────────────────

const BASE: NetWorthInputs = {
  coins: 1_000,
  botStakes: { 'bot-A': { staked: 200 } },
  holdings: [{ ticker: 'MEME', units: 10, price: 50 }], // portfolio = 500

describe('computeNetWorth (Patch 16.3)', () => {

  // Test 2 — zero liabilities

    const inputs: NetWorthInputs = { ...BASE, liabilities: [] };

    const inputs: NetWorthInputs = { ...BASE, botStakes: {} };
```
