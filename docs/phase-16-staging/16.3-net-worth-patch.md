## Blocker Context

The HUD currently shows a static seed number for portfolio/net-worth — it never reflects actual game state.
This patch introduces `src/logic/netWorth.ts` as a pure computation module, wires it into `sleep()`,
and replaces the static HUD number with a live Zustand selector + a 28-day sparkline.

**Integration dependency:**
- **Requires Patch 16.2** — `botStakes: Record<string, { staked: number }>` must exist in the store.
- `src/logic/portfolio.ts` must export `aggregate(holdings): { netWorth: number }`.
- `src/logic/debtDungeon.ts` must expose or the store must expose the `liabilities` array. If liabilities live only in `debtDungeon.ts`, pass them in from the component layer; if in the store, read from state directly inside `sleep()`.

## 2. New pure module — `src/logic/netWorth.ts`

```typescript
// src/logic/netWorth.ts
// Pure — zero store imports. All inputs passed explicitly.

import { aggregate } from './portfolio';
import type { NetWorthInputs } from '../types/netWorth';

/**
 * Computes the player's net worth at a point in time.
 *
 * Formula:
 *   netWorth = coins
 *            + Σ botStakes[].staked          (locked capital, not yet settled)
 *            + portfolio.aggregate(holdings).netWorth
 *            - Σ liabilities[].balance
 *
 * Returns a number; can be negative (liabilities exceed assets).
 */
export function computeNetWorth(inputs: NetWorthInputs): number {
  const { coins, botStakes, holdings, liabilities } = inputs;

  const stakedTotal = Object.values(botStakes).reduce(
    (sum, entry) => sum + entry.staked,
    0,
  );

  const portfolioValue = holdings.length > 0
    ? aggregate(holdings).netWorth
    : 0;

  const liabilityTotal = liabilities.reduce(
    (sum, l) => sum + l.balance,
    0,
  );

  return coins + stakedTotal + portfolioValue - liabilityTotal;
}
```

### 3d. Integrate into `sleep()` — append after bot settlement delta (Patch 16.2)

```typescript
// After bot settlement coins adjustment:
set((state) => {
  const inputs: NetWorthInputs = {
    coins: state.coins,
    botStakes: state.botStakes,
    holdings: state.holdings ?? [],      // adjust field name to match your store shape
    liabilities: state.liabilities ?? [],
  };

  const nw = computeNetWorth(inputs);

  const history = [...state.netWorthHistory, nw];
  // Keep only the most recent 28 entries (one per sleep = one per in-game day)
  const trimmedHistory = history.length > 28 ? history.slice(history.length - 28) : history;

  return {
    netWorth: nw,
    netWorthHistory: trimmedHistory,
  };
});
```

## 4. HUD component patch — `src/components/HudOverlay.tsx`

Replace the static portfolio/net-worth number and add a sparkline.

```tsx
// src/components/HudOverlay.tsx  (patch — do not replace entire file)

import { useStore } from '../state/store';

// --- inside HudOverlay component ---

const netWorth = useStore((s) => s.netWorth);
const history = useStore((s) => s.netWorthHistory);

// -----------------------------------------------------------------------
// Sparkline helper — 30×16 px SVG, min-max normalized, stroke var(--sim-lawn)
// -----------------------------------------------------------------------
function NetWorthSparkline({ data }: { data: number[] }) {
  if (data.length < 2) return null;

  const W = 30;
  const H = 16;
  const min = Math.min(...data);
  const max = Math.max(...data);
  const range = max - min || 1; // avoid /0 when all values equal

  const points = data
    .map((v, i) => {
      const x = (i / (data.length - 1)) * W;
      const y = H - ((v - min) / range) * H;
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    })
    .join(' ');

  return (
    <svg
      width={W}
      height={H}
      viewBox={`0 0 ${W} ${H}`}
      aria-label="Net worth trend"
      style={{ display: 'inline-block', verticalAlign: 'middle', marginLeft: 4 }}
    >
      <polyline
        points={points}
        fill="none"
        stroke="var(--sim-lawn)"
        strokeWidth="1.5"
        strokeLinejoin="round"
        strokeLinecap="round"
      />
    </svg>
  );
}

// -----------------------------------------------------------------------
// Replace the static seed render with:
// -----------------------------------------------------------------------
return (
  // ... existing HUD JSX ...
  <span className="hud-net-worth">
    {netWorth.toLocaleString('th-TH', { maximumFractionDigits: 0 })} 🪙
    <NetWorthSparkline data={history} />
  </span>
  // ...
);
```

## 5. New test file — `src/logic/net-worth.test.ts`

```typescript
import { describe, it, expect } from 'vitest';
import { computeNetWorth } from './netWorth';
import type { NetWorthInputs } from '../types/netWorth';
import { createStore } from '../state/store';

// ---------------------------------------------------------------------------
// Fixtures
// ---------------------------------------------------------------------------

const BASE_INPUTS: NetWorthInputs = {
  coins: 1000,
  botStakes: { 'bot-A': { staked: 200 } },
  holdings: [{ ticker: 'MEME', units: 10, price: 50 }], // netWorth = 500
  liabilities: [{ id: 'l1', nameTh: 'บัตรเครดิต', balance: 300, apr: 0.18, minMonthly: 50 }],
};

// ---------------------------------------------------------------------------
// Tests
// ---------------------------------------------------------------------------

describe('computeNetWorth (patch 16.3)', () => {
  it('golden-value: coins + stakes + portfolio - liabilities', () => {
    // 1000 + 200 + 500 - 300 = 1400
    expect(computeNetWorth(BASE_INPUTS)).toBe(1400);
  });

  it('zero liabilities case — no subtraction', () => {
    const inputs: NetWorthInputs = { ...BASE_INPUTS, liabilities: [] };
    // 1000 + 200 + 500 = 1700
    expect(computeNetWorth(inputs)).toBe(1700);
  });

  it('zero botStakes case — staked total is 0', () => {
    const inputs: NetWorthInputs = { ...BASE_INPUTS, botStakes: {} };
    // 1000 + 0 + 500 - 300 = 1200
    expect(computeNetWorth(inputs)).toBe(1200);
  });

  it('history caps at 28 after 29th sleep', () => {
    const store = createStore();
    store.setState({
      coins: 500,
      botStakes: {},
      holdings: [],
      liabilities: [],
      netWorthHistory: Array(28).fill(500), // already full
    });

    store.getState().sleep();

    expect(store.getState().netWorthHistory.length).toBe(28);
    // The oldest entry (index 0 before sleep) is dropped, newest appended
    expect(store.getState().netWorthHistory[27]).toBe(store.getState().netWorth);
  });

  it('history appends one entry per sleep', () => {
    const store = createStore();
    store.setState({ coins: 500, botStakes: {}, holdings: [], liabilities: [], netWorthHistory: [] });

    store.getState().sleep();
    expect(store.getState().netWorthHistory.length).toBe(1);

    store.getState().sleep();
    expect(store.getState().netWorthHistory.length).toBe(2);
  });

  it('netWorth in store matches computeNetWorth output after sleep', () => {
    const store = createStore();
    store.setState({ coins: 800, botStakes: {}, holdings: [], liabilities: [], netWorthHistory: [] });

    store.getState().sleep();

    const expected = computeNetWorth({
      coins: store.getState().coins,
      botStakes: store.getState().botStakes,
      holdings: (store.getState() as any).holdings ?? [],
      liabilities: (store.getState() as any).liabilities ?? [],
    });
    expect(store.getState().netWorth).toBe(expected);
  });

  it('HUD renders live netWorth number from store (component smoke test)', async () => {
    // Uses @testing-library/react — adjust import to project convention
    const { render } = await import('@testing-library/react');
    const { HudOverlay } = await import('../components/HudOverlay');

    const store = createStore();
    store.setState({ netWorth: 42_000, netWorthHistory: [40000, 41000, 42000] });

    // Provide store via context or patch module mock per your project pattern
    const { getByText } = render(<HudOverlay />);
    expect(getByText(/42,000/)).toBeTruthy();
  });

  it('netWorth can be negative when liabilities exceed all assets', () => {
    const inputs: NetWorthInputs = {
      coins: 0,
      botStakes: {},
      holdings: [],
      liabilities: [{ id: 'l1', nameTh: 'หนี้', balance: 9999, apr: 0.25, minMonthly: 200 }],
    };
    expect(computeNetWorth(inputs)).toBe(-9999);
  });
});
```
