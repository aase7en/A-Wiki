## Ticket Goal

Un-orphan `src/logic/botSettlement.ts`. The player stakes farm coins into a configured bot;
at `sleep()` the day's mock P&L settles back into coins. Loss is strictly bounded to the staked
amount (already guaranteed by `botSettlement.ts`). Converts bot investing from cosmetic sparkline
into a real economic consequence.

## 5. `stakeBot` Action

```typescript
// src/state/store.ts  — inside create() / immer produce / set() block

stakeBot(deviceId: string, amount: number): void {
  set((state) => {
    // Reject if below minimum stake
    if (amount < STAKE_MIN) {
      console.warn(
        `[stakeBot] rejected: amount ${amount} is below STAKE_MIN ${STAKE_MIN}`
      );
      return;
    }

    // Clamp to cap — never stake more than STAKE_CAP per bot
    const clamped = Math.min(amount, STAKE_CAP);

    // Clamp to available coins — never go negative
    const actual = Math.min(clamped, state.coins);
    if (actual <= 0) return;

    state.coins = Math.max(0, state.coins - actual);
    state.botStakes[deviceId] = { staked: actual };
  });
},
```

---

## 6. `unstakeBot` Action

Settles the final epoch P&L using the bot's current equity curve, removes the stake entry, and
returns coins to the player balance.

```typescript
// src/state/store.ts  — inside create() / immer produce / set() block

unstakeBot(deviceId: string): void {
  set((state) => {
    const stakeEntry = state.botStakes[deviceId];
    if (!stakeEntry) return;

    const bot = state.bots[deviceId];
    const curve = bot?.equityCurve ?? [];

    // settle() returns net P&L; negative losses are bounded to -staked
    const pnl = settle(stakeEntry.staked, curve);
    const returned = stakeEntry.staked + pnl;

    state.coins = Math.max(0, state.coins + returned);
    delete state.botStakes[deviceId];
  });
},
```

## 7. `sleep()` Integration

Inside the existing `sleep()` action, **after** the bot epoch refresh loop, add the settlement
pass. The settlement loop iterates every active `botStakes` entry, calls `settle()`, and credits /
debits coins:

```typescript
// src/state/store.ts  — inside sleep() action
// Place this block AFTER the existing bot epoch refresh / equityCurve update:

// ── Bot stake settlement (Patch 16.2) ────────────────────────────────────────
set((state) => {
  for (const [deviceId, stakeEntry] of Object.entries(state.botStakes)) {
    const bot = state.bots[deviceId];
    if (!bot) continue; // bot removed mid-session — silently skip

    // settle() returns net P&L for this epoch
    // Loss is bounded to -staked by botSettlement.ts guarantees
    const pnl = settle(stakeEntry.staked, bot.equityCurve);

    // Credit P&L; loss is bounded upstream but clamp coins defensively
    state.coins = Math.max(0, state.coins + pnl);

    // Update staked principal: loss erodes the staked amount toward 0
    const newStaked = Math.max(0, stakeEntry.staked + pnl);
    if (newStaked === 0) {
      delete state.botStakes[deviceId]; // fully liquidated
    } else {
      state.botStakes[deviceId].staked = newStaked;
    }
  }
});
// ── End bot stake settlement ──────────────────────────────────────────────────
```

## 9. New Test File — `src/state/bot-settle.test.ts`

```typescript
// src/state/bot-settle.test.ts
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createStore } from './store';           // adjust path to match project
import { settle, STAKE_MIN, STAKE_CAP } from '../logic/botSettlement';

// ── Mock settle() so tests control P&L deterministically ─────────────────────
vi.mock('../logic/botSettlement', () => ({
  STAKE_MIN: 50,
  STAKE_CAP: 2_000,
  settle: vi.fn(),
}));

const mockSettle = vi.mocked(settle);

// ── Fixtures ─────────────────────────────────────────────────────────────────

function makeBot(equityCurve: number[] = [100, 105, 103]): any {
  return { equityCurve, epochDay: 1, deviceId: 'bot1', name: 'TestBot' };
}

function freshStore(coins: number, botEquityCurve: number[] = [100, 105, 103]) {
  const store = createStore();
  store.setState({
    coins,
    bots: { bot1: makeBot(botEquityCurve) },
    botStakes: {},
  });
  return store;
}

// ── Tests ─────────────────────────────────────────────────────────────────────

describe('Bot Settlement (Patch 16.2)', () => {
  beforeEach(() => {
    mockSettle.mockReset();
    mockSettle.mockReturnValue(0); // default: flat P&L
  });

  // Test 1 — staking deducts coins
  it('staking deducts coins from balance', () => {
    const store = freshStore(500);
    store.getState().stakeBot('bot1', 200);

    expect(store.getState().coins).toBe(300);
    expect(store.getState().botStakes['bot1'].staked).toBe(200);
  });

  // Test 2 — below STAKE_MIN rejected
  it('rejects stake below STAKE_MIN — coins and botStakes unchanged', () => {
    const store = freshStore(500);
    store.getState().stakeBot('bot1', STAKE_MIN - 1); // 49 coins

    expect(store.getState().coins).toBe(500);         // unchanged
    expect(store.getState().botStakes['bot1']).toBeUndefined();
  });

  // Test 3 — above STAKE_CAP clamped
  it('clamps stake to STAKE_CAP when requested amount exceeds cap', () => {
    const store = freshStore(9_000);
    store.getState().stakeBot('bot1', STAKE_CAP + 999); // request 2999

    expect(store.getState().botStakes['bot1'].staked).toBe(STAKE_CAP); // 2000
    expect(store.getState().coins).toBe(9_000 - STAKE_CAP);            // 7000
  });

  // Test 4 — profitable day credits coins
  it('profitable sleep() credits P&L to coins', () => {
    mockSettle.mockReturnValue(75); // +75 P&L

    const store = freshStore(500);
    store.getState().stakeBot('bot1', 100); // coins → 400
    store.getState().sleep();

    // coins after sleep = 400 + 75 = 475
    expect(store.getState().coins).toBeGreaterThanOrEqual(475);
  });

  // Test 5 — losing day never drives coins below 0
  it('losing sleep() never drives coins below zero', () => {
    mockSettle.mockReturnValue(-200); // max loss = -staked

    const store = freshStore(200);
    store.getState().stakeBot('bot1', 200); // coins → 0
    store.getState().sleep();

    expect(store.getState().coins).toBeGreaterThanOrEqual(0);
  });

  // Test 6 — deterministic after save/reload
  it('settlement is deterministic after save and reload from SAVEABLE_FIELDS', () => {
    mockSettle.mockReturnValue(30);

    const storeA = freshStore(600);
    storeA.getState().stakeBot('bot1', 100);

    // Capture saveable snapshot
    const snapshot = {
      coins: storeA.getState().coins,
      bots: storeA.getState().bots,
      botStakes: storeA.getState().botStakes,
    };

    // Reload into two independent stores and sleep both
    const storeB = createStore();
    storeB.setState(snapshot);
    storeB.getState().sleep();

    const storeC = createStore();
    storeC.setState(snapshot);
    storeC.getState().sleep();

    expect(storeB.getState().coins).toBe(storeC.getState().coins);
  });

  // Test 7 — unstake returns settled balance
  it('unstakeBot settles P&L and returns full balance to coins', () => {
    mockSettle.mockReturnValue(40); // +40 P&L on unstake

    const store = freshStore(500);
    store.getState().stakeBot('bot1', 150); // coins → 350

    store.getState().unstakeBot('bot1');

    // coins = 350 (remaining) + 150 (principal back) + 40 (P&L) = 540
    expect(store.getState().coins).toBe(540);
    expect(store.getState().botStakes['bot1']).toBeUndefined();
  });

  // Test 8 — arbitrage guard: re-staking same bot replaces, no double-deduction
  it('arbitrage guard: re-staking same bot replaces position (no infinite coins)', () => {
    const store = freshStore(1_000);
    store.getState().stakeBot('bot1', 200); // coins → 800, staked 200

    // Stake again on same bot — must overwrite, not stack
    store.getState().stakeBot('bot1', 300); // if replace: coins → 500, staked 300
                                            // if additive: coins → 500, staked 500 (bug)

    const { coins, botStakes } = store.getState();
    const staked = botStakes['bot1']?.staked ?? 0;

    // Core invariant: total deducted from coins === total currently staked
    expect(1_000 - coins).toBe(staked);
    expect(coins).toBeGreaterThanOrEqual(0);
  });
});
```
