## Blocker Context

`src/logic/botSettlement.ts` exists and exposes `settle(stake, equityCurve): number` (net P&L for the epoch).
Constants `STAKE_MIN = 50` and `STAKE_CAP = 2000` are defined there.
Currently nothing in the store calls `settle()` — bot investing is purely cosmetic sparklines.
This patch makes it a real economy: coins leave the player's wallet when they stake, and come back (with gain or bounded loss) on `sleep()`.

**Integration dependency:** None — this is the root patch. Patches 16.3, 16.4, 16.5 all depend on `botStakes` being in the store as established here.

### 2d. Add `stakeBot` action

```typescript
stakeBot: (deviceId: string, amount: number) => {
  set((state) => {
    if (amount < STAKE_MIN) {
      // Reject: below minimum — caller should surface toast
      return {};
    }

    const effectiveAmount = Math.min(amount, STAKE_CAP);
    const coinsAfter = state.coins - effectiveAmount;

    if (coinsAfter < 0) {
      // Reject: not enough coins
      return {};
    }

    return {
      coins: Math.max(0, coinsAfter),
      botStakes: {
        ...state.botStakes,
        [deviceId]: { staked: effectiveAmount },
      },
    };
  });
},
```

### 2e. Add `unstakeBot` action

```typescript
unstakeBot: (deviceId: string) => {
  set((state) => {
    const entry = state.botStakes[deviceId];
    if (!entry) return {};

    const bot = state.bots[deviceId];
    const pnl = bot ? settle(entry.staked, bot.equityCurve) : 0;
    const settledCoins = entry.staked + pnl;

    const remaining = { ...state.botStakes };
    delete remaining[deviceId];

    return {
      coins: Math.max(0, state.coins + settledCoins),
      botStakes: remaining,
    };
  });
},
```

### 2f. Integrate into `sleep()` — add after existing bot epoch refresh

```typescript
// Inside the sleep() action, after bots are refreshed for the new epoch:
set((state) => {
  let coinsDelta = 0;

  for (const [deviceId, stakeEntry] of Object.entries(state.botStakes)) {
    const bot = state.bots[deviceId];
    if (!bot) continue;

    // settle() returns net P&L; loss is bounded to staked amount by botSettlement.ts guarantees
    const pnl = settle(stakeEntry.staked, bot.equityCurve);
    coinsDelta += pnl;
  }

  return {
    coins: Math.max(0, state.coins + coinsDelta),
  };
});
```

## 3. New test file — `src/state/bot-settle.test.ts`

```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import { createStore } from './store';          // adjust import path as needed
import { STAKE_MIN, STAKE_CAP } from '../logic/botSettlement';

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

/** Builds a minimal bot state with a predictable equity curve */
function makeBotWithCurve(equityCurve: number[]): import('../types/store').BotState {
  return {
    equityCurve,
    // fill other required BotState fields with safe defaults:
    deviceId: 'bot-test',
    name: 'TestBot',
    // ... add any other required fields your BotState type demands
  } as any;
}

function freshStore(coins = 1000) {
  const store = createStore();
  store.setState({ coins, bots: {}, botStakes: {} });
  return store;
}

// ---------------------------------------------------------------------------
// Tests
// ---------------------------------------------------------------------------

describe('Bot Settlement (patch 16.2)', () => {
  it('staking deducts coins from balance', () => {
    const store = freshStore(500);
    store.getState().stakeBot('bot-A', 200);
    expect(store.getState().coins).toBe(300);
    expect(store.getState().botStakes['bot-A']?.staked).toBe(200);
  });

  it('staking below STAKE_MIN is rejected — coins unchanged', () => {
    const store = freshStore(500);
    store.getState().stakeBot('bot-A', STAKE_MIN - 1);
    expect(store.getState().coins).toBe(500);
    expect(store.getState().botStakes['bot-A']).toBeUndefined();
  });

  it('staking above STAKE_CAP clamps to STAKE_CAP', () => {
    const store = freshStore(5000);
    store.getState().stakeBot('bot-A', STAKE_CAP + 500);
    expect(store.getState().botStakes['bot-A']?.staked).toBe(STAKE_CAP);
    expect(store.getState().coins).toBe(5000 - STAKE_CAP);
  });

  it('profitable sleep() credits correct P&L to coins', () => {
    const store = freshStore(500);
    // Inject a bot with a rising equity curve so settle() returns positive P&L
    const gainCurve = [100, 105, 110, 115, 120]; // +20 gain on 100 staked = +20%
    store.setState({
      bots: { 'bot-A': makeBotWithCurve(gainCurve) },
    });
    store.getState().stakeBot('bot-A', 100);
    const coinsBeforeSleep = store.getState().coins; // 400

    store.getState().sleep();

    expect(store.getState().coins).toBeGreaterThan(coinsBeforeSleep);
  });

  it('losing sleep() never drives coins below 0', () => {
    const store = freshStore(50); // barely enough to stake minimum
    const lossCurve = [100, 90, 80, 70, 60]; // loss scenario
    store.setState({
      bots: { 'bot-A': makeBotWithCurve(lossCurve) },
    });
    store.getState().stakeBot('bot-A', STAKE_MIN); // stakes all 50
    // coins = 0 before sleep; even with negative P&L, floor at 0
    store.getState().sleep();
    expect(store.getState().coins).toBeGreaterThanOrEqual(0);
  });

  it('settlement is deterministic after save/reload', () => {
    const store1 = freshStore(500);
    const gainCurve = [100, 102, 104, 106];
    store1.setState({ bots: { 'bot-A': makeBotWithCurve(gainCurve) } });
    store1.getState().stakeBot('bot-A', 100);
    store1.getState().sleep();
    const coinsAfter1 = store1.getState().coins;

    // Simulate save/reload by cloning saveable state into a fresh store
    const savedState = {
      coins: 400, // pre-sleep coins
      botStakes: { 'bot-A': { staked: 100 } },
      bots: { 'bot-A': makeBotWithCurve(gainCurve) },
    };
    const store2 = createStore();
    store2.setState(savedState);
    store2.getState().sleep();

    expect(store2.getState().coins).toBe(coinsAfter1);
  });

  it('unstakeBot returns settled balance to coins', () => {
    const store = freshStore(500);
    const flatCurve = [100, 100, 100, 100]; // zero P&L
    store.setState({ bots: { 'bot-A': makeBotWithCurve(flatCurve) } });
    store.getState().stakeBot('bot-A', 200);
    expect(store.getState().coins).toBe(300);

    store.getState().unstakeBot('bot-A');

    // P&L is ~0 on flat curve; staked 200 should be returned
    expect(store.getState().coins).toBeCloseTo(500, 0);
    expect(store.getState().botStakes['bot-A']).toBeUndefined();
  });

  it('arbitrage guard: staking same bot twice does not create infinite coins', () => {
    const store = freshStore(1000);
    const curve = [100, 105];
    store.setState({ bots: { 'bot-A': makeBotWithCurve(curve) } });

    // First stake
    store.getState().stakeBot('bot-A', 200);
    const coinsAfterFirst = store.getState().coins; // 800

    // Second stake on same bot — should overwrite or reject, never add coins
    store.getState().stakeBot('bot-A', 200);
    const coinsAfterSecond = store.getState().coins;

    // Coins must not exceed what we started with minus one stake's worth
    expect(coinsAfterSecond).toBeLessThanOrEqual(coinsAfterFirst);
    expect(store.getState().coins).toBeGreaterThanOrEqual(0);
  });
});
```
