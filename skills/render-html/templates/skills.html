var D = window.AWIKI_DATA; var skills = (D.skills || []).slice(); var summary = D.summary || {}; function esc(s){ return String(s==null?"":s).replace(/[&<>]/g,function(c){return {"&":"&","<":"<",">":">"}[c];}); } var worst = summary.FAIL>0 ? "fail" : (summary.WARN>0 ? "warn" : "ok"); document.getElementById("summary").innerHTML = ''+worst.toUpperCase()+''+ 'Skill Quality'+ ''+ ''+(summary.OK||0)+' OK'+ ''+(summary.WARN||0)+' WARN'+ ''+(summary.FAIL||0)+' FAIL'; var order={FAIL:0,WARN:1,OK:2}; var activeFilter="all", sortKey=null, sortDir=1; function rows(){ var rs = skills.filter(function(s){ return activeFilter==="all" || s.level===activeFilter; }); rs.sort(function(a,b){ if(sortKey){ var x=a[sortKey], y=b[sortKey]; if(typeof x==="number"&&typeof y==="number") return (x-y)*sortDir; return String(x).localeCompare(String(y))*sortDir; } // default: worst level first return (order[a.level]==null?9:order[a.level]) - (order[b.level]==null?9:order[b.level]); }); return rs; } function render(){ var tb=document.querySelector("#tbl tbody"); tb.innerHTML=""; var rs=rows(); rs.forEach(function(s){ var lvl=(s.level||"").toLowerCase(); var issues=(s.issues||[]).map(function(i){ return ''+esc(i)+''; }).join("") || '—'; var tr=document.createElement("tr"); tr.innerHTML = ''+esc(s.level)+''+ ''+esc(s.name)+''+ ''+issues+''+ ''+esc(s.chars)+''+ ''+(s.eval_covered?'✅':'—')+''+ ''+esc(s.path)+''; tb.appendChild(tr); }); document.getElementById("count").textContent = rs.length+" / "+skills.length+" skills"; } document.querySelectorAll("#filters button").forEach(function(b){ b.addEventListener("click", function(){ document.querySelectorAll("#filters button").forEach(function(x){ x.classList.remove("active"); }); b.classList.add("active"); activeFilter=b.getAttribute("data-f"); render(); }); }); document.querySelectorAll("#tbl th").forEach(function(th){ th.style.cursor="pointer"; th.addEventListener("click", function(){ var k=th.getAttribute("data-k"); sortDir=(sortKey===k)?-sortDir:1; sortKey=k; render(); }); }); window.AWIKI_EXPORT = function(){ return { needs_fix: skills.filter(function(s){ return s.level!=="OK"; }).map(function(s){ return {name:s.name, level:s.level, issues:s.issues}; }) }; }; render();
