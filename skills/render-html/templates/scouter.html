Math.round(n/1000)+"K" : n; } function role(id){ return recoIds[id]; } var sortKey=null, sortDir=1, query=""; function rows(){ var q=query.toLowerCase(); return models.filter(function(m){ return !q || COLS.some(function(c){ return String(m[c]||"").toLowerCase().indexOf(q)>=0; }); }).sort(function(a,b){ if(!sortKey) return 0; var x=a[sortKey], y=b[sortKey]; if(typeof x==="number"&&typeof y==="number") return (x-y)*sortDir; return String(x).localeCompare(String(y))*sortDir; }); } function render(){ var tb=document.querySelector("#tbl tbody"); tb.innerHTML=""; var rs=rows(); rs.forEach(function(m){ var tr=document.createElement("tr"); var r=role(m.id); if(r) tr.style.background="color-mix(in srgb, var(--accent) 14%, transparent)"; tr.innerHTML = ''+m.id+''+(r?'
